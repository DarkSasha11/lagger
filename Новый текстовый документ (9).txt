-- // ACCURATE SCANNER //
local function Scan()
    local target = nil
    for _, obj in ipairs(game:GetDescendants()) do
        if obj:IsA("RemoteEvent") then
            local path = obj:GetFullName()
            local bad = false
            for _, b in ipairs(blacklistedNames) do
                if string.find(path, b, 1, true) or string.find(obj.Name, b, 1, true) then 
                    bad = true 
                    break 
                end
            end
            if not bad then target = obj break end
        end
    end
    State.Target = target
    if target then
        StatusText.Text = "CONNECTED: "..target.Name
        StatusText.TextColor3 = Color3.new(0,1,0.6)
    else
        StatusText.Text = "NOT SUPPORTED"
        StatusText.TextColor3 = Color3.new(1,0,0)
        Tw(Stroke, {Color = Color3.new(1,0,0)})
        CoreStroke.Color = Color3.new(0.3,0,0)
        CoreIcon.ImageColor3 = Color3.new(0.3,0,0)
        PowerBtn.BackgroundColor3 = Color3.new(0,0,0)
        BindLabel.Text = "[ SYSTEM LOCKED ]"
        BindLabel.TextColor3 = Color3.new(0.5,0,0)
    end
end

-- // INSTANT STOP ENGINE //
RunService.Heartbeat:Connect(function()
    if not State.LagEnabled or not State.Target then return end
    
    for i = 1, State.Packets do
        if not State.LagEnabled then break end
        pcall(function() State.Target:FireServer(State.UUID, Payload) end)
    end
end)
