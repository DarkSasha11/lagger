local function Scan()
    local target = nil
    for _, obj in ipairs(game:GetDescendants()) do
        if obj:IsA("RemoteEvent") then
            local path = obj:GetFullName()
            local bad = false
            for _, b in ipairs(blacklistedNames) do
                -- Проверка наличия части строки в полном пути или имени
                if string.find(path, b, 1, true) then bad = true break end
            end
            if not bad then target = obj break end
        end
    end
    State.Target = target
    if target then
        StatusText.Text = "> LINKED: "..target.Name
        StatusText.TextColor3 = Color3.fromRGB(0,255,150)
    else
        StatusText.Text = "> !!! NOT SUPPORTED !!!"
        StatusText.TextColor3 = Color3.new(1,0,0)
        CoreStroke.Color = Color3.new(0.2,0,0)
        CoreIcon.ImageColor3 = Color3.new(0.2,0,0)
        Tw(Stroke, {Color = Color3.new(1,0,0)})
        PowerBtn.BackgroundColor3 = Color3.new(0,0,0)
    end
end
